## From Windows PowerShell Cookbook (O'Reilly)

Set-StrictMode -Version Latest

if($message.Body -match "book")
{
    [Console]::WriteLine("This is a message about the book.")
}
else
{
    [Console]::WriteLine("This is an unknown message.")
}
