function Send-Text {
   param(
      [Parameter(ValueFromPipeline=$true)]$InputObject = $global:WatinIE
   ,
      [Parameter(Position=0)]
      [string]$text
   ,
      [switch]$passthru
   ,
      [switch]$append
   )
   process {
      if($append) {
         $InputObject.AppendText( $text )
      } else {
         $InputObject.TypeText( $text )
      }
      if($Passthru){$InputObject}
   }
}
