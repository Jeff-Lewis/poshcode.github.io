# mountvol_fix.cmd

### [download](5652.txt)



```text
@echo off
  setlocal enabledelayedexpansion
    set "key=HKLM\SYSTEM\MountedDevices"
    for /f "tokens=1,3" %%i in (
      'reg query %key% ^| findstr /irc:"dos" ^| sort'
    ) do (
      for /f "tokens=1,3" %%k in (
        'reg query %key% ^| find "%%j" ^| find /v /i "dos"'
      ) do (
        set "dsk=%%i"
        if "%%j" equ "%%l" echo !dsk:~-2! -^> %%k
      )
    )
  endlocal
exit /b
```
