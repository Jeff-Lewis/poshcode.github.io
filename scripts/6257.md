function Speak-Text
{
    [CmdletBinding()]
    [OutputType([int])]
    Param
    (
        # Param1 help description
        [Parameter(Mandatory=$true,
                   ValueFromPipelineByPropertyName=$true,
                   Position=0)]
        [string]$text
    )

    Begin
    {
        $speaker = New-Object -com SAPI.SpVoice
    }
    Process
    {
        $speaker.Speak($text)
    }
    End
    {
        $speaker.Dispose()
    }
}
